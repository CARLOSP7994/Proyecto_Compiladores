%{
/* Necesita esto para la llamada a atof () abajo */
#include <math.h>
%}

DIGIT    [0-9]
ID       [a-z][a-z0-9]*

%%

{DIGIT}+    {
            printf( "Entero: %s (%d)\n", yytext,
                    atoi( yytext ) );
            }

{DIGIT}+"."{DIGIT}*        {
            printf( "flotante: %s (%g)\n", yytext,
                    atof( yytext ) );
            }

if|then|begin|end|procedure|function        {
            printf( "Palabra clave: %s\n", yytext );
            }

{ID}        printf( "Identificador: %s\n", yytext );

"+"|"-"|"*"|"/"   printf( "Operador: %s\n", yytext );

"{"[^}\n]*"}" / * omite línea Comentarios * /     

[ \t\n]+ / * omite el espacio en blanco * /          

.           printf( "Caracter no especificado: %s\n", yytext );

%%

main( argc, argv )
int argc;
char **argv;
    {
    ++argv, --argc; 
    if ( argc > 0 )
            yyin = fopen( argv[0], "r" );
    else
            yyin = stdin;

    yylex();
    }
